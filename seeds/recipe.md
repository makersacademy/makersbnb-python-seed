CREATE TABLE Users (
  user id SERIAL PRIMARY KEY,
  username text,
  name text,
  password varchar(255),
  email,
  phone number
);

  title text,

-- The foreign key name is always {other_table_singular}_id
  artist_id int,
  constraint fk_artist foreign key(artist_id)
    references artists(id)
    on delete cascade
